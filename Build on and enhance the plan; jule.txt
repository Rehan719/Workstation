import { createContext, useContext, useState } from 'react';

interface UserProfile {
  preferences: {
    theme: 'light' | 'dark' | 'islamic';
    language: string;
    accessibility: { highContrast: boolean; fontSize: string };
  };
  emotionalState?: 'happy' | 'frustrated' | 'neutral';
}

const UserProfileContext = createContext<{
  profile: UserProfile;
  updateProfile: (updates: Partial<UserProfile>) => void;
} | undefined>(undefined);

export const UserProfileProvider: React.FC = ({ children }) => {
  const [profile, setProfile] = useState<UserProfile>({
    preferences: { theme: 'light', language: 'en', accessibility: { highContrast: false, fontSize: 'medium' } },
  });

  const updateProfile = (updates: Partial<UserProfile>) => setProfile((prev) => ({ ...prev, ...updates }));
  return <UserProfileContext.Provider value={{ profile, updateProfile }}>{children}</UserProfileContext.Provider>;
};

export const useUserProfile = () => {
  const context = useContext(UserProfileContext);
  if (!context) throw new Error('useUserProfile must be used within UserProfileProvider');
  return context;
};

        if (data.action === 'navigate') window.location.href = data.target;
        else if (data.action === 'settings') {
          // Handle settings update (e.g., theme change)
          console.log('Settings update:', data);
        }
        setResponseText(`Executed: ${command}`);
      } catch (error) {
        setResponseText('Sorry, I couldn’t process that command.');
      }
    };
    recognition.start();
  };

  return (
    <div className="p-4 bg-gray-100 rounded-lg">
      <button onClick={handleVoiceCommand} className="px-4 py-2 bg-blue-500 text-white rounded">
        {t('speak')}
      </button>
      <p className="mt-2">{responseText}</p>
    </div>
  );
};

import { Kafka } from 'kafkajs';
import { AntColony } from 'ant-colony-optimization';
const consumer = new Kafka({ clientId:'swarm', brokers:['localhost:9092'] })
  .consumer({ groupId:'swarm-group' });
export class SwarmIntelligenceService {
  static async getGroupInsights() {
    await consumer.connect();
    await consumer.subscribe({ topic:'user-data' });
    const data:any[] = [];
    await consumer.run({ eachMessage: ({ message }) =>
      data.push(JSON.parse(message.value!.toString()))
    });
    return new AntColony(data).optimize();
  }
}

import { Kafka } from 'kafkajs';
import { AntColony } from 'ant-colony-optimization';
const consumer = new Kafka({ clientId:'swarm', brokers:['localhost:9092'] })
  .consumer({ groupId:'swarm-group' });
export class SwarmIntelligenceService {
  static async getGroupInsights() {
    await consumer.connect();
    await consumer.subscribe({ topic:'user-data' });
    const data:any[] = [];
    await consumer.run({ eachMessage: ({ message }) =>
      data.push(JSON.parse(message.value!.toString()))
    });
    return new AntColony(data).optimize();
  }
}
